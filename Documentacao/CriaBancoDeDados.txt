 `alcunhaCidadao` varchar(50) CHARACTER SET utf8 NOT NULL,
 `rgCidadao` varchar(10) CHARACTER SET utf8 NOT NULL,
 `dataCidadao` date NOT NULL,
 `nacionalidadeCidadao` varchar(50) CHARACTER SET utf8 NOT NULL,
 `estadoCidadao` varchar(50) CHARACTER SET utf8 NOT NULL,
 `cidadeCidadao` varchar(50) CHARACTER SET utf8 NOT NULL,
 `ruaCidadao` varchar(50) CHARACTER SET latin1 COLLATE latin1_spanish_ci NOT NULL,
 `numeroCidadao` int(11) NOT NULL,
 `bairroCidadao` varchar(50) CHARACTER SET utf8 NOT NULL,
 PRIMARY KEY (`cpfCidadao`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1
